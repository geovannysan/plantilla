SELECT 'Enero' as MES, 'GUAYAS' AS Provicia, 'GUAYAQUIL' AS CANTON,u.nombre AS 'NOMBRE DE USUARIO', u.direccion_principal,u.movil AS 'TELEFONO', '0' AS 'Numero', 'SPEED' AS 'EMPRESA','INALAMBRICO' AS 'TIPO', CAST(SUBSTRING_INDEX(p.velocidad, '/', 1) AS UNSIGNED) AS Up, CAST(SUBSTRING_INDEX(p.velocidad, '/', 1) AS UNSIGNED) AS Down, p.plan, ROUND((p.costo)* 1.15,2) as 'valor con iva' 
FROM usuarios u 
INNER JOIN tblservicios t ON t.idcliente = u.id 
INNER JOIN perfiles p ON t.idperfil = p.id 
WHERE u.estado = 'ACTIVO'
